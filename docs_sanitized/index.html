    BasicSetup BoundaryConditions NetCDFData HybridApproaches

    BasicThermalConvection

    Startwithasimplethermalconvectionsimulation:

    usingGeodynamo

functionbasic_thermal_convection()
#1.Createconfiguration
config=create_optimized_config(32,32,nlat=64,nlon=128)

#2.Setupradialdomain(Earth-like)
domain=create_radial_domain(0.35,1.0,64)

#3.Createtemperaturefield
temp_field=create_shtns_temperature_field(Float64,config,domain)

#4.Simpleboundaries:hotbottom,coldtop
temp_boundaries=create_hybrid_temperature_boundaries(
(:uniform,4000.0),#HotCMB(4000K)
(:uniform,300.0),#Coolsurface(300K)
config
)

#5.Applyboundaries
apply_netcdf_temperature_boundaries!(temp_field,temp_boundaries)

println("Basicthermalconvectionsetupcomplete!")
end

    ProgrammaticBoundaryPatterns

    Createrealisticboundaryconditionsusingbuilt-inpatterns:

    #HotplumeatCMB
temp_boundaries=create_hybrid_temperature_boundaries(
(:plume,4200.0,Dict(
"center_theta"=>π/3,#Plumelocation(60°fromnorth)
"center_phi"=>0.0,#Primemeridian
"width"=>π/6#30°plumewidth
)),
(:uniform,300.0),#Uniformcoolsurface
config
)

#Sphericalharmonicpatterns
y11_boundaries=create_hybrid_temperature_boundaries(
(:y11,4000.0,Dict("amplitude"=>200.0)),#Y₁₁+perturbation
(:uniform,300.0),#Uniformsurface
config
)

#Availablepatterns::uniform,:y11,:y20,:plume,:hemisphere,
#:dipole,:quadrupole,:checkerboard,:custom

    UsingNetCDFDataFiles

    Loadboundaryconditionsfromobservationalormodeldata:

    #1.CreatesampleNetCDFfiles(one-timesetup)
include("examples/create_sample_netcdf_boundaries.jl")

#2.LoadtemperatureboundariesfromNetCDFfiles
temp_boundaries=load_temperature_boundaries("cmb_temp.nc","surface_temp.nc")

#3.Loadcompositionalboundaries
comp_boundaries=load_composition_boundaries("cmb_comp.nc","surface_comp.nc")

#4.Applytofields
apply_netcdf_temperature_boundaries!(temp_field,temp_boundaries)
apply_netcdf_composition_boundaries!(comp_field,comp_boundaries)

#5.Inspectloadeddata
print_boundary_info(temp_boundaries)
stats=get_boundary_statistics(temp_boundaries.inner_boundary)
println("CMBtemperaturerange:[$(stats["min"]),$(stats["max"])]K")

    NetCDFFileFormat

    YourNetCDFfilesshouldinclude:

    • Coordinates:theta(colatitude)andphi(longitude)
    • Datavariable:temperature,composition,etc.
    • Optional:timedimensionfortime-dependentboundaries
    • Metadata:units,descriptions,andattributes

    HybridApproaches

    MixNetCDFdatawithprogrammaticpatternsformaximumflexibility:

    #NetCDFinnerboundary+programmaticouterboundary
temp_boundaries=create_hybrid_temperature_boundaries(
"realistic_cmb.nc",#ComplexCMBfromdatafile
(:uniform,300.0),#Simpleuniformsurface
config
)

#Programmaticinner+NetCDFouter
temp_boundaries=create_hybrid_temperature_boundaries(
(:plume,4200.0,Dict("width"=>π/6)),#Analyticalplumepattern
"earth_surface_temps.nc",#Realsurfacetemperaturedata
config
)

#Customfunctionexample
my_pattern(theta,phi)=sin(3*theta)*cos(2*phi)+0.5*cos(theta)^2

custom_boundaries=create_hybrid_temperature_boundaries(
(:custom,4000.0,Dict("function"=>my_pattern)),#Yourfunction
(:uniform,300.0),#Simpleouter
config
)

    WhyHybrid?

    • Flexibility:Usedatawhereavailable,analyticalwherenot
    • Performance:Simpleboundarieswherephysicsallows
    • Testing:Comparecomplexvs.simpleboundaryeffects
    • Development:Graduallyaddcomplexitytoboundaries
