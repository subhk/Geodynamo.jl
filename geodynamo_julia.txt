module PencilSetup
    using MPI
    using PencilArrays
    using PencilFFTs
    using ..Parameters
    
    # MPI communicator
    const comm = MPI.COMM_WORLD
    
    # Pencil decomposition for (θ, φ, r) data
    # Primary decomposition along φ (azimuthal) direction for FFTs
    function create_pencil_topology()
        # Initialize MPI
        MPI.Init()
        
        # Get MPI info
        rank = MPI.Comm_rank(comm)
        nprocs = MPI.Comm_size(comm)
        
        # Create 2D process grid: (P_θ, P_φ)
        # Optimal decomposition depends on problem size and number of processes
        P_θ = isqrt(nprocs)
        P_φ = nprocs ÷ P_θ
        
        if rank == 0
            println("MPI setup: $nprocs processes, grid: $P_θ × $P_φ")
        end
        
        # Create pencil decomposition
        # Dimensions: (θ, φ, r)
        dims = (i_Th, i_Ph, i_N)
        
        # First pencil: distributed along θ (for Legendre transforms)
        pencil_θ = Pencil(dims, (1,), comm)
        
        # Second pencil: distributed along φ (for FFTs) 
        pencil_φ = Pencil(dims, (2,), comm)
        
        # Third pencil: distributed along r (for radial operations)
        pencil_r = Pencil(dims, (3,), comm)
        
        return pencil_θ, pencil_φ, pencil_r
    end
    
    # Create transforms between pencils
    function create_transforms(pencil_θ, pencil_φ, pencil_r)
        # Transform θ-pencil → φ-pencil (for FFTs)
        transform_θ_to_φ = PencilArray(pencil_θ) => PencilArray(pencil_φ)
        
        # Transform φ-pencil → r-pencil (for radial operations)
        transform_φ_to_r = PencilArray(pencil_φ) => PencilArray(pencil_r)
        
        # Reverse transforms
        transform_φ_to_θ = PencilArray(pencil_φ) => PencilArray(pencil_θ)
        transform_r_to_φ = PencilArray(pencil_r .) => PencilArray(pencil_φ)
        
        return transform_θ_to_φ, transform_φ_to_r, transform_φ_to_θ, transform_r_to_φ
    end
    
    # FFT plans for azimuthal direction
    function create_fft_plans(pencil_φ)
        # Create PencilArray for complex data
        u_real = PencilArray{Float64}(undef, pencil_φ)
        u_complex = PencilArray{ComplexF64}(undef, pencil_φ)
        
        # FFT along φ direction (dimension 2)
        fft_plan = PencilFFTPlan(u_real, u_complex, 2)
        ifft_plan = PencilFFTPlan(u_complex, u_real, 2, FFTW.BACKWARD)
        
        return fft_plan, ifft_plan, u_real, u_complex
    end
    
    export create_pencil_topology, create_transforms, create_fft_plans, comm
end

module VariableTypes
    using PencilArrays
    using StaticArrays
    using ..Parameters
    using ..PencilSetup
    
    # Harmonic structure for spherical harmonics
    struct HarmonicModes
        l::Vector{Int}            # Spherical harmonic degree
        m::Vector{Int}            # Azimuthal wavenumber
        n_modes::Int              # Total number of modes
        local_modes::UnitRange{Int}  # Local mode range for this process
    end
    
    # Radial domain structure
    struct RadialDomain
        N::Int                    # Total number of points
        local_range::UnitRange{Int}  # Local radial points
        r::Matrix{Float64}        # Radial points r^p for p=-3:3
        dr_matrices::Vector{Matrix{Float64}}  # Derivative matrices
        radial_laplacian::Matrix{Float64}     # d²/dr² + (2/r)d/dr
        integration_weights::Vector{Float64}   # ∫ r² dr weights
    end
    
    # Spectral field in (l,m,r) space
    struct SpectralField{T<:Number}
        modes::HarmonicModes
        domain::RadialDomain
        data_real::PencilArray{T,3}   # Real part
        data_imag::PencilArray{T,3}   # Imaginary part
    end
    
    # Physical field in (θ,φ,r) space
    struct PhysicalField{T<:Number}
        pencil_θ::Pencil{3}
        pencil_φ::Pencil{3} 
        pencil_r::Pencil{3}
        data_θ::PencilArray{T,3}      # Data distributed along θ
        data_φ::PencilArray{T,3}      # Data distributed along φ
        data_r::PencilArray{T,3}      # Data distributed along r
    end
    
    # Vector field with 3 components
    struct VectorField{T<:Number}
        r_component::PhysicalField{T}
        θ_component::PhysicalField{T}
        φ_component::PhysicalField{T}
    end
    
    # Toroidal-Poloidal decomposition
    struct TorPolField{T<:Number}
        toroidal::SpectralField{T}
        poloidal::SpectralField{T}
    end
    
    # Constructor functions
    function create_harmonic_modes(pencil_r)
        l_vals = Int[]
        m_vals = Int[]
        
        for l in 0:i_L1
            for m in 0:min(l, i_M1)
                push!(l_vals, l)
                push!(m_vals, m)
            end
        end
        
        n_modes = length(l_vals)
        # Distribute modes across processes (simplified)
        local_modes = 1:n_modes  # This should be properly distributed
        
        return HarmonicModes(l_vals, m_vals, n_modes, local_modes)
    end
    
    function create_radial_domain(pencil_r)
        # Get local radial range for this process
        local_range = pencil_r.axes[3]
        N = i_N
        
        # Initialize radial points (Chebyshev-Gauss-Lobatto)
        r = zeros(N, 7)  # r^p for p=-3:3
        for n in 1:N
            r[n, 4] = 0.5 * (1.0 + cos(π * (N - n) / (N - 1)))  # r^1
        end
        
        # Map to spherical shell
        ri = d_rratio / (1.0 - d_rratio)
        r[:, 4] .+= ri
        
        # Compute other powers
        for p in 1:7
            if p != 4
                power = p - 4
                r[:, p] = r[:, 4] .^ power
            end
        end
        
        # Initialize matrices (simplified - full computation needed)
        dr_matrices = [zeros(2*i_KL+1, N) for _ in 1:3]
        radial_laplacian = zeros(2*i_KL+1, N)
        integration_weights = zeros(N)
        
        return RadialDomain(N, local_range, r, dr_matrices, radial_laplacian, integration_weights)
    end
    
    function create_spectral_field(::Type{T}, modes::HarmonicModes, domain::RadialDomain, 
                                  pencil_r::Pencil{3}) where T
        # Create pencil arrays for spectral data
        dims = (modes.n_modes, 1, domain.N)  # (modes, dummy, radius)
        pencil_spec = Pencil(dims, (3,), PencilSetup.comm)  # Distributed along radius
        
        data_real = PencilArray{T}(undef, pencil_spec)
        data_imag = PencilArray{T}(undef, pencil_spec)
        
        fill!(data_real, zero(T))
        fill!(data_imag, zero(T))
        
        return SpectralField{T}(modes, domain, data_real, data_imag)
    end
    
    function create_physical_field(::Type{T}, pencil_θ, pencil_φ, pencil_r) where T
        data_θ = PencilArray{T}(undef, pencil_θ)
        data_φ = PencilArray{T}(undef, pencil_φ)
        data_r = PencilArray{T}(undef, pencil_r)
        
        fill!(data_θ, zero(T))
        fill!(data_φ, zero(T))
        fill!(data_r, zero(T))
        
        return PhysicalField{T}(pencil_θ, pencil_φ, pencil_r, data_θ, data_φ, data_r)
    end
    
    export HarmonicModes, RadialDomain, SpectralField, PhysicalField, VectorField, TorPolField
    export create_harmonic_modes, create_radial_domain, create_spectral_field, create_physical_field
end

    # Legendre polynomial evaluation
    struct LegendreData
        P::Matrix{Float64}        # Associated Legendre functions P_l^m(cos θ)
        dP::Matrix{Float64}       # Derivatives dP/dθ

    end
    
    function initialize_legendre()
        # Gauss-Legendre quadrature points and weights
        theta = Vector{Float64}(undef, i_Th)
        weights = Vector{Float64}(undef, i_Th)
        
        # Compute Gauss points on [-1,1], then map to [0,π]
        x, w = gausslegendre(i_Th)
        theta = acos.(-x)  # Map to [0,π]
        weights = w
        
        sin_theta = sin.(theta)
        cos_theta = cos.(theta)
        
        # Compute associated Legendre functions
        n_modes = (i_L1 + 1) * (i_L1 + 2) ÷ 2
        P = zeros(i_Th, n_modes)
        dP = zeros(i_Th, n_modes)

                end
                idx += 1

    # Spherical harmonic transforms
    function spectral_to_physical!(spec::SpectralField{T}, phys::PhysicalField{T}, 
                                  legendre::LegendreData, fft_plan, transform_ops) where T
        # Step 1: Transform from spectral (l,m,r) to intermediate (θ,m,r)
        intermediate = similar(phys.data_θ)
        
        # Legendre synthesis: sum over l for each m
        @views for r_idx in axes(spec.data_real, 3)
            for m in 0:i_M1
                for θ_idx in 1:i_Th
                    real_sum = zero(T)
                    imag_sum = zero(T)
                    
                    # Sum over l for this m
                    for l in m:i_L1
                        mode_idx = get_mode_index(l, m, spec.modes)
                        if mode_idx !== nothing
                            real_sum += spec.data_real[mode_idx, 1, r_idx] * legendre.P[θ_idx, mode_idx]
                            imag_sum += spec.data_imag[mode_idx, 1, r_idx] * legendre.P[θ_idx, mode_idx]
                        end
                    end
                    
                    # Store in intermediate array (complex for FFT)
                    intermediate[θ_idx, m+1, r_idx] = complex(real_sum, imag_sum)
                end
            end
        end
        
        # Step 2: FFT along φ direction to get (θ,φ,r)
        transpose!(phys.data_φ, intermediate, transform_ops.θ_to_φ)
        mul!(phys.data_φ, fft_plan, phys.data_φ)
        
        # Take real part if needed
        if T <: Real
            phys.data_φ .= real.(phys.data_φ)
        end
    end
    
    function physical_to_spectral!(phys::PhysicalField{T}, spec::SpectralField{T},
                                  legendre::LegendreData, ifft_plan, transform_ops) where T
        # Step 1: IFFT along φ to get (θ,m,r)
        intermediate = similar(phys.data_φ)
        mul!(intermediate, ifft_plan, phys.data_φ)
        
        # Step 2: Legendre analysis: integrate over θ for each m
        @views for r_idx in axes(spec.data_real, 3)
            for mode_idx in spec.modes.local_modes
                l, m = spec.modes.l[mode_idx], spec.modes.m[mode_idx]
                
                real_integral = zero(T)
                imag_integral = zero(T)
                
                for θ_idx in 1:i_Th
                    weight = legendre.weights[θ_idx] * sin(legendre.theta[θ_idx])
                    P_val = legendre.P[θ_idx, mode_idx]
                    
                    real_integral += weight * P_val * real(intermediate[θ_idx, m+1, r_idx])
                    imag_integral += weight * P_val * imag(intermediate[θ_idx, m+1, r_idx])
                end
                
                # Normalization
                norm_factor = (2*l + 1) / (4*π)
                if m > 0
                    norm_factor *= 2
                end
                
                spec.data_real[mode_idx, 1, r_idx] = real_integral * norm_factor
                spec.data_imag[mode_idx, 1, r_idx] = imag_integral * norm_factor
                
                # m=0 modes are real
                if m == 0
                    spec.data_imag[mode_idx, 1, r_idx] = zero(T)
                end
            end
        end
    end
    
    function get_mode_index(l::Int, m::Int, modes::HarmonicModes)
        for i in modes.local_modes
            if modes.l[i] == l && modes.m[i] == m
                return i
            end
        end
        return nothing
    end

module LinearOps
    using LinearAlgebra
    using SparseArrays
    using PencilArrays
    using ..Parameters
    using ..VariableTypes
    
    # Banded matrix structure for radial derivatives
    struct BandedMatrix{T}
        data::Matrix{T}           # Banded storage
        bandwidth::Int            # Half-bandwidth
        size::Int                 # Matrix size
    end
    
    function create_derivative_matrix(order::Int, domain::RadialDomain)
        # Create finite difference matrix for given derivative order
        N = domain.N
        bandwidth = i_KL
        
        # Initialize banded matrix storage
        data = zeros(2*bandwidth + 1, N)
        
        # Compute finite difference coefficients using Chebyshev points
        for n in 1:N
            left = max(1, n - bandwidth)
            right = min(N, n + bandwidth)
            stencil_size = right - left + 1
            
            # Vandermonde matrix for interpolation
            V = ones(stencil_size, stencil_size)
            points = domain.r[left:right, 4]  # r values
            
            for j in 2:stencil_size
                for i in 1:stencil_size
                    V[i, j] = V[i, j-1] * (points[i] - domain.r[n, 4])
                end
            end
            
            # Solve for derivative coefficients
            rhs = zeros(stencil_size)
            if order <= stencil_size
                rhs[order + 1] = factorial(order)
            end
            
            coeffs = V \ rhs
            
            # Store in banded format
            for (i, idx) in enumerate(left:right)
                band_row = bandwidth + 1 + n - idx
                if 1 <= band_row <= 2*bandwidth + 1
                    data[band_row, idx] = coeffs[i]
                end
            end
        end
        
        return BandedMatrix(data, bandwidth, N)
    end
    
    function create_radial_laplacian(domain::RadialDomain)
        # d²/dr² + (2/r) d/dr
        d2_matrix = create_derivative_matrix(2, domain)
        d1_matrix = create_derivative_matrix(1, domain)
        
        laplacian_data = copy(d2_matrix.data)
        
        # Add (2/r) * d/dr term
        for n in 1:domain.N
            r_inv = domain.r[n, 3]  # 1/r
            for j in max(1, n - i_KL):min(domain.N, n + i_KL)
                band_row = i_KL + 1 + n - j
                laplacian_data[band_row, j] += 2.0 * r_inv * d1_matrix.data[band_row, j]
            end
        end
        
        return BandedMatrix(laplacian_data, i_KL, domain.N)
    end
    
    # Apply banded matrix to pencil array
    function apply_banded_matrix!(output::PencilArray{T,3}, matrix::BandedMatrix{T}, 
                                 input::PencilArray{T,3}) where T
        # Apply matrix along radial direction (axis 3)
        @views for spec_idx in axes(input, 1), dummy_idx in axes(input, 2)
            apply_banded_vector!(output[spec_idx, dummy_idx, :], matrix, input[spec_idx, dummy_idx, :])
        end
    end
    
    function apply_banded_vector!(output::AbstractVector{T}, matrix::BandedMatrix{T}, 
                                 input::AbstractVector{T}) where T
        fill!(output, zero(T))
        N = matrix.size
        bandwidth = matrix.bandwidth
        
        for j in 1:N
            for i in max(1, j - bandwidth):min(N, j + bandwidth)
                band_row = bandwidth + 1 + i - j
                output[i] += matrix.data[band_row, j] * input[j]
            end
        end
    end
    
    export BandedMatrix, create_derivative_matrix, create_radial_laplacian, apply_banded_matrix!
end

module Timestepping
    using LinearAlgebra
    using PencilArrays
    using ..Parameters
    using ..VariableTypes
    using ..LinearOps
    
    # Timestepping state
    mutable struct TimestepState
        time::Float64
        dt::Float64
        step::Int
        iteration::Int
        error::Float64
        converged::Bool
    end
    
    # Implicit matrices for each spherical harmonic mode
    struct ImplicitMatrices{T}
        matrices::Vector{BandedMatrix{T}}  # One for each l value
        factorizations::Vector{Any}       # LU factorizations
    end
    
    function create_timestepping_matrices(modes::HarmonicModes, domain::RadialDomain, 
                                        diffusivity::Float64, dt::Float64)
        matrices = BandedMatrix{Float64}[]
        factorizations = Any[]
        
        # Create matrix for each l value
        for l in 0:i_L1
            # Implicit operator: (1/dt - diffusivity * implicit_factor * L_l)
            # where L_l = d²/dr² + (2/r)d/dr - l(l+1)/r²
            
            laplacian = create_radial_laplacian(domain)
            
            # Modify for spherical harmonic l
            l_factor = Float64(l * (l + 1))
            implicit_data = copy(laplacian.data)
            
            # Time derivative term
            implicit_data[i_KL + 1, :] .+= 1.0 / dt
            
            # Diffusion term  
            implicit_data .*= -diffusivity * d_implicit
            
            # Spherical harmonic term
            for n in 1:domain.N
                r_inv_sq = domain.r[n, 2]  # 1/r²
                implicit_data[i_KL + 1, n] += diffusivity * d_implicit * l_factor * r_inv_sq
            end
            
            matrix = BandedMatrix(implicit_data, i_KL, domain.N)
            push!(matrices, matrix)
            
            # Create LU factorization (simplified - use proper banded solver)
            dense_matrix = banded_to_dense(matrix)
            factorization = lu(dense_matrix)
            push!(factorizations, factorization)
        end
        
        return ImplicitMatrices(matrices, factorizations)
    end
    
    function banded_to_dense(matrix::BandedMatrix{T}) where T
        # Convert banded matrix to dense for LU factorization
        N = matrix.size
        bandwidth = matrix.bandwidth
        dense = zeros(T, N, N)
        
        for j in 1:N
            for i in max(1, j - bandwidth):min(N, j + bandwidth)
                band_row = bandwidth + 1 + i - j
                dense[i, j] = matrix.data[band_row, j]
            end
        end
        
        return dense
    end
    
    function apply_explicit_operator!(output::SpectralField{T}, input::SpectralField{T},
                                     nonlinear::SpectralField{T}, domain::RadialDomain,
                                     diffusivity::Float64, dt::Float64) where T
        # Explicit operator: (1/dt + diffusivity * (1-implicit_factor) * L_l) * input + nonlinear
        
        @views for mode_idx in input.modes.local_modes
            l = input.modes.l[mode_idx]
            
            # Create explicit operator for this l
            laplacian = create_radial_laplacian(domain)
            l_factor = Float64(l * (l + 1))
            
            # Apply to input
            for r_idx in 1:domain.N
                # Time derivative
                output.data_real[mode_idx, 1, r_idx] = input.data_real[mode_idx, 1, r_idx] / dt
                output.data_imag[mode_idx, 1, r_idx] = input.data_imag[mode_idx, 1, r_idx] / dt
                
                # Explicit diffusion (simplified application)
                # This should use proper banded matrix multiplication
                diffusion_real = zero(T)
                diffusion_imag = zero(T)
                
                # Add diffusion terms (placeholder)
                output.data_real[mode_idx, 1, r_idx] += diffusion_real
                output.data_imag[mode_idx, 1, r_idx] += diffusion_imag
                
                # Add nonlinear terms
                output.data_real[mode_idx, 1, r_idx] += nonlinear.data_real[mode_idx, 1, r_idx]
                output.data_imag[mode_idx, 1, r_idx] += nonlinear.data_imag[mode_idx, 1, r_idx]
            end
        end
    end
    
    function solve_implicit_step!(solution::SpectralField{T}, rhs::SpectralField{T},
                                 matrices::ImplicitMatrices{T}) where T
        # Solve implicit system for each mode
        @views for mode_idx in solution.modes.local_modes
            l = solution.modes.l[mode_idx]
            factorization = matrices.factorizations[l + 1]
            
            # Solve for real part
            rhs_real = rhs.data_real[mode_idx, 1, :]
            solution.data_real[mode_idx, 1, :] = factorization \ rhs_real
            
            # Solve for imaginary part  
            rhs_imag = rhs.data_imag[mode_idx, 1, :]
            solution.data_imag[mode_idx, 1, :] = factorization \ rhs_imag
        end
    end
    
    function compute_timestep_error(new_field::SpectralField{T}, old_field::SpectralField{T}) where T
        error = zero(Float64)
        
        for mode_idx in new_field.modes.local_modes
            for r_idx in 1:new_field.domain.N
                diff_real = new_field.data_real[mode_idx, 1, r_idx] - old_field.data_real[mode_idx, 1, r_idx]
                diff_imag = new_field.data_imag[mode_idx, 1, r_idx] - old_field.data_imag[mode_idx, 1, r_idx]
                error += diff_real^2 + diff_imag^2
            end
        end
        
        # Global reduction across all processes
        global_error = MPI.Allreduce(error, MPI.SUM, PencilSetup.comm)
        return sqrt(global_error)
    end
    
    export TimestepState, ImplicitMatrices, create_timestepping_matrices
    export apply_explicit_operator!, solve_implicit_step!, compute_timestep_error
end

        vort_θ = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        vort_φ = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        vorticity = VectorField(vort_r, vort_θ, vort_φ)
        
        # Spectral fields
        toroidal = create_spectral_field(T, modes, domain, pencil_r)
        poloidal = create_spectral_field(T, modes, domain, pencil_r)
        nl_toroidal = create_spectral_field(T, modes, domain, pencil_r)
        nl_poloidal = create_spectral_field(T, modes, domain, pencil_r)
        pressure = create_spectral_field(T, modes, domain, pencil_r)

                                        temp_field, comp_field, mag_field,
                                        legendre_data, transform_ops) where T

        compute_advection_term!(fields)
        
        # Add Coriolis force: -2Ω × u
        add_coriolis_force!(fields)
        
        # Add buoyancy forces
        add_thermal_buoyancy!(fields, temp_field)
        add_compositional_buoyancy!(fields, comp_field)
        
        # Add Lorentz force: (∇ × B) × B

        # Transform to spectral space

    end
    
    function compute_advection_term!(fields::VelocityFields{T}) where T
        # Compute u × ω in physical space
        vel = fields.velocity
        vort = fields.vorticity
        
        # u × ω = (u_θ ω_φ - u_φ ω_θ, u_φ ω_r - u_r ω_φ, u_r ω_θ - u_θ ω_r)
        @. vel.r_component.data_φ = d_Ro * (vel.θ_component.data_φ * vort.φ_component.data_φ - 
                                           vel.φ_component.data_φ * vort.θ_component.data_φ)
        @. vel.θ_component.data_φ = d_Ro * (vel.φ_component.data_φ * vort.r_component.data_φ - 
                                           vel.r_component.data_φ * vort.φ_component.data_φ)
        @. vel.φ_component.data_φ = d_Ro * (vel.r_component.data_φ * vort.θ_component.data_φ - 
                                           vel.θ_component.data_φ * vort.r_component.data_φ)
    end

module MagneticField
    using PencilArrays
    using ..Parameters
    using ..VariableTypes
    using ..Timestepping
    using ..LinearOps
    using ..PencilSetup
    
    struct MagneticFields{T}
        # Physical space magnetic field
        magnetic::VectorField{T}
        current::VectorField{T}
        
        # Spectral representation
        toroidal::SpectralField{T}
        poloidal::SpectralField{T}
        
        # Inner core fields
        ic_toroidal::SpectralField{T}
        ic_poloidal::SpectralField{T}
        
        # Nonlinear terms (induction)
        nl_toroidal::SpectralField{T}
        nl_poloidal::SpectralField{T}
        
        # Imposed field (if any)
        imposed_field::Union{VectorField{T}, Nothing}
    end
    
    function create_magnetic_fields(::Type{T}, modes, domain_oc, domain_ic, pencils) where T
        pencil_θ, pencil_φ, pencil_r = pencils
        
        # Physical space fields
        mag_r = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        mag_θ = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        mag_φ = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        magnetic = VectorField(mag_r, mag_θ, mag_φ)
        
        curr_r = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        curr_θ = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        curr_φ = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        current = VectorField(curr_r, curr_θ, curr_φ)
        
        # Spectral fields
        toroidal = create_spectral_field(T, modes, domain_oc, pencil_r)
        poloidal = create_spectral_field(T, modes, domain_oc, pencil_r)
        
        # Inner core fields (different domain)
        ic_toroidal = create_spectral_field(T, modes, domain_ic, pencil_r)
        ic_poloidal = create_spectral_field(T, modes, domain_ic, pencil_r)
        
        # Nonlinear terms
        nl_toroidal = create_spectral_field(T, modes, domain_oc, pencil_r)
        nl_poloidal = create_spectral_field(T, modes, domain_oc, pencil_r)
        
        imposed_field = nothing
        
        return MagneticFields{T}(magnetic, current, toroidal, poloidal,
                                ic_toroidal, ic_poloidal, nl_toroidal, nl_poloidal,
                                imposed_field)
    end
    
    function compute_magnetic_nonlinear!(mag_fields::MagneticFields{T}, 
                                        vel_fields, rotation_rate,
                                        legendre_data, transform_ops) where T
        # Compute induction equation: ∂B/∂t = ∇ × (u × B) + η∇²B
        
        # Compute u × B (electromotive force)
        compute_induction_term!(mag_fields, vel_fields)
        
        # Inner core rotation effects
        add_inner_core_rotation!(mag_fields, rotation_rate)
        
        # Transform to spectral space
        physical_to_spectral!(mag_fields.magnetic, mag_fields.nl_toroidal, mag_fields.nl_poloidal,
                             legendre_data, transform_ops)
    end
    
    function compute_induction_term!(mag_fields::MagneticFields{T}, vel_fields) where T
        # Compute u × B in physical space
        vel = vel_fields.velocity
        mag = mag_fields.magnetic
        
        # u × B = (u_θ B_φ - u_φ B_θ, u_φ B_r - u_r B_φ, u_r B_θ - u_θ B_r)
        emf_r = similar(mag.r_component.data_φ)
        emf_θ = similar(mag.θ_component.data_φ)
        emf_φ = similar(mag.φ_component.data_φ)
        
        @. emf_r = vel.θ_component.data_φ * mag.φ_component.data_φ - 
                   vel.φ_component.data_φ * mag.θ_component.data_φ
        @. emf_θ = vel.φ_component.data_φ * mag.r_component.data_φ - 
                   vel.r_component.data_φ * mag.φ_component.data_φ
        @. emf_φ = vel.r_component.data_φ * mag.θ_component.data_φ - 
                   vel.θ_component.data_φ * mag.r_component.data_φ
        
        # Store in magnetic field structure for curl computation
        mag.r_component.data_φ .= emf_r
        mag.θ_component.data_φ .= emf_θ
        mag.φ_component.data_φ .= emf_φ
    end
    
    function add_inner_core_rotation!(mag_fields::MagneticFields{T}, Ω::Float64) where T
        # Inner core rotation: -Ω × B_ic
        # This affects the boundary conditions and coupling
        
        # Rotation effects on inner core field (placeholder)

        @. mag_fields.ic_poloidal.data_real *= (1.0 - Ω * 1e-3)  # Simplified
    end
    
    export MagneticFields, create_magnetic_fields, compute_magnetic_nonlinear!
end

        # Nonlinear terms (advection)
        nonlinear::SpectralField{T}
        
        # Sources and boundary conditions
        internal_sources::Vector{T}
        boundary_values::Matrix{T}
    end
    
    function create_temperature_field(::Type{T}, modes, domain, pencils) where T
        pencil_θ, pencil_φ, pencil_r = pencils
        
        # Temperature field
        temperature = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        
        # Gradient components
        grad_r = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        grad_θ = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        grad_φ = create_physical_field(T, pencil_θ, pencil_φ, pencil_r)
        gradient = VectorField(grad_r, grad_θ, grad_φ)
        
        # Spectral representation
        spectral = create_spectral_field(T, modes, domain, pencil_r)
        nonlinear = create_spectral_field(T, modes, domain, pencil_r)
        
        # Sources and boundary conditions

module GeodynamoSimulation
    using MPI
    using PencilArrays
    using PencilFFTs
    using HDF5
    using ..Parameters
    using ..PencilSetup
    using ..VariableTypes
    using ..SphericalTransforms
    using ..Timestepping
    using ..Velocity
    using ..MagneticField
    using ..Temperature
    
    # Main simulation state
    struct SimulationState{T}
        # Field variables
        velocity::VelocityFields{T}
        magnetic::MagneticFields{T}
        temperature::TemperatureField{T}
        
        # Geometric data
        harmonic_modes::HarmonicModes
        radial_domain::RadialDomain
        inner_core_domain::RadialDomain
        legendre_data::LegendreData
        
        # Pencil decomposition
        pencils::Tuple{Pencil{3}, Pencil{3}, Pencil{3}}
        transforms::NamedTuple
        fft_plans::NamedTuple
        
        # Timestepping
        timestep_state::TimestepState
        implicit_matrices::Dict{Symbol, ImplicitMatrices{T}}
        
        # I/O
        output_counter::Int
    end
    
    function initialize_simulation(::Type{T} = Float64) where T
        # Initialize MPI and pencil decomposition
        pencil_θ, pencil_φ, pencil_r = create_pencil_topology()
        
        # Create transform operations
        transform_θ_to_φ, transform_φ_to_r, transform_φ_to_θ, transform_r_to_φ = 
            create_transforms(pencil_θ, pencil_φ, pencil_r)
        
        # Create FFT plans
        fft_plan, ifft_plan, u_real, u_complex = create_fft_plans(pencil_φ)
        
        transforms = (θ_to_φ = transform_θ_to_φ, φ_to_r = transform_φ_to_r,
                     φ_to_θ = transform_φ_to_θ, r_to_φ = transform_r_to_φ)
        fft_plans = (forward = fft_plan, backward = ifft_plan, 
                    real_array = u_real, complex_array = u_complex)
        
        # Initialize geometric data
        harmonic_modes = create_harmonic_modes(pencil_r)
        radial_domain = create_radial_domain(pencil_r)
        inner_core_domain = create_radial_domain(pencil_r)  # Modify for inner core
        legendre_data = initialize_legendre()
        
        # Create field variables
        pencils = (pencil_θ, pencil_φ, pencil_r)
        velocity = create_velocity_fields(T, harmonic_modes, radial_domain, pencils)
        magnetic = create_magnetic_fields(T, harmonic_modes, radial_domain, 
                                        inner_core_domain, pencils)
        temperature = create_temperature_field(T, harmonic_modes, radial_domain, pencils)
        
        # Initialize timestepping
        timestep_state = TimestepState(d_time, d_timestep, 0, 0, Inf, false)
        
        # Create implicit matrices for each equation
        implicit_matrices = Dict{Symbol, ImplicitMatrices{T}}()
        implicit_matrices[:velocity] = create_timestepping_matrices(
            harmonic_modes, radial_domain, d_E, d_timestep)
        implicit_matrices[:magnetic] = create_timestepping_matrices(
            harmonic_modes, radial_domain, 1.0/d_Pm, d_timestep)
        implicit_matrices[:temperature] = create_timestepping_matrices(
            harmonic_modes, radial_domain, 1.0/d_Pr, d_timestep)
        
        return SimulationState{T}(velocity, magnetic, temperature, harmonic_modes,
                                 radial_domain, inner_core_domain, legendre_data,
                                 pencils, transforms, fft_plans, timestep_state,
                                 implicit_matrices, 0)
    end
    
    function run_simulation!(state::SimulationState{T}) where T
        rank = MPI.Comm_rank(comm)
        
        if rank == 0
            println("Starting geodynamo simulation with PencilArrays...")
            println("Grid: $(i_Th) × $(i_Ph) × $(i_N)")
            println("Harmonic modes: $(state.harmonic_modes.n_modes)")
        end
        
        # Main timestepping loop
        while state.timestep_state.step < i_maxtstep && 
              state.timestep_state.time < 1.0
            
            # Predictor-corrector iterations
            state.timestep_state.iteration = 1
            state.timestep_state.error = Inf
            state.timestep_state.converged = false
            
            while state.timestep_state.iteration <= 10 && 
                  state.timestep_state.error > d_dterr
                
                # Compute nonlinear terms
                compute_all_nonlinear_terms!(state)
                
                # Timestepping (predictor or corrector)
                if state.timestep_state.iteration == 1
                    predictor_step!(state)
                else
                    corrector_step!(state)
                end
                
                # Transform to physical space for next iteration
                transform_to_physical!(state)
                
                # Check convergence
                state.timestep_state.iteration += 1
                # Error computation would be done in the stepping functions
            end
            
            # Advance time
            state.timestep_state.time += state.timestep_state.dt
            state.timestep_state.step += 1
            
            # Adaptive timestepping
            compute_cfl_timestep!(state)
            
            # Output
            if state.timestep_state.step % i_save_rate2 == 0
                output_fields!(state)
                
                if rank == 0
                    println("Step: $(state.timestep_state.step), " *
                           "Time: $(state.timestep_state.time), " *
                           "dt: $(state.timestep_state.dt)")
                end
            end
        end
        
        if rank == 0
            println("Simulation completed!")
        end
    end
